 create or replace trigger $brName$
 	before delete or insert or update
 	on $targetTable$
 	for each row

 declare
 	l_oper			varchar2(3);
 	l_error_stach		varchar2(4000;
 begin
 	if inserting
 	then
  		l_oper := 'INS';
  	elsif updating
  	then
  		l_oper := 'UPD';
  	elsif deleting
  	then
  	l_oper := 'DEL';
  end if;
